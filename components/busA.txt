library ieee; 
 use ieee.std_logic_1164.all; 
 use IEEE.std_logic_signed.all; 

---sigin = [MDR,GR]

entity busA is  
     port( 
          
	 clock:in std_logic(3 downto 0);
	 sigin:in std_logic_vector(3 downto 0);
	 busA_out:out std_logic_vector(3 downto 0);
	);
end busA; 

---architecture

architecture BEHAVIOR of busA is
begin
if(sigin="01")then
	busA_out =< GR;
elseif(sigin="10")then
	busA_out =< MDR;

end busA;
